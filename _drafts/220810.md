def next_permutation(nums):
    i = len(nums) - 2
    j = len(nums) - 1

    while nums[i] > nums[i + 1]:
        i -= 1        
    while nums[i] > nums[j]:
        j -= 1

    nums[i], nums[j] = nums[j], nums[i]

    return nums[0:i+1] + list(reversed(nums[i+1:len(nums)]))
        

def prev_permutation(nums):
    i = len(nums) - 2
    j = len(nums) - 1

    while nums[i] < nums[i + 1]:
        i -= 1
    while nums[i] < nums[j]:
        j -= 1

    nums[i], nums[j] = nums[j], nums[i]

    return nums[0:i+1] + list(reversed(nums[i+1:len(nums)]))
